MySQL Queries, please execute in the same order.

1. CREATE SCHEMA TESLA;

2. USE TESLA;

3. CREATE TABLE CAR(
	VIN VARCHAR(100) NOT NULL PRIMARY KEY,
	MAKE VARCHAR(50),
	MODEL VARCHAR(50),
	YEAR VARCHAR(30)
 );

4. CREATE TABLE RATING(
	ID INT AUTO_INCREMENT PRIMARY KEY,
    REVIEWEDBY VARCHAR(100),
    SAFETY INT,
    PERFORMANCE INT,
    TECHNOLOGY INT,
    INTERIOR INT,
    RELIABILITY INT,
    VIN VARCHAR(100),
    FOREIGN KEY (VIN) REFERENCES CAR(VIN)
 );

5. INSERT INTO CAR(VIN, MAKE, MODEL, YEAR)
    VALUES('S1','Tesla', 'S', '2016');

6. INSERT INTO CAR(VIN, MAKE, MODEL, YEAR)
    VALUES('X1','Tesla', 'X', '2017');

    VALUES('Edmunds',5,5,5,5,5,'S1');
